rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Allow read access to all files
    match /{allPaths=**} {
      allow read: if true;
    }
    
    // Allow authenticated users to upload to avatars folder
    match /avatars/{fileName} {
      allow write: if request.auth != null;
    }
  }
}
